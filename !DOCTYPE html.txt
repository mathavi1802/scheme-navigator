  Aadhaar Number: <input type="text" name="aadhaar" required><br>
  Age: <input type="number" name="age" required><br>
  Income: <input type="number" name="income" required><br>
  Occupation: <input type="text" name="occupation" required><br>
  <button type="submit">Check Schemes</button>
</form>
<h2>Eligible Schemes:</h2>
<ul id="result"></ul>

<script>
document.getElementById("schemeForm").addEventListener("submit", async (e) => {
  e.preventDefault();
  const formData = new FormData(e.target);
  const data = {
    aadhaar: formData.get("aadhaar"),
    age: parseInt(formData.get("age")),
    income: parseFloat(formData.get("income")),
    occupation: formData.get("occupation")
  };
  
  try {
    const response = await fetch("http://127.0.0.1:8000/get-schemes", {
      method: "POST",
      headers: {"Content-Type": "application/json"},
      body: JSON.stringify(data)
    });

    if (!response.ok) {
      const err = await response.json();
      throw new Error(err.detail);
    }

    const result = await response.json();
    const ul = document.getElementById("result");
    ul.innerHTML = "";
    result.eligible_schemes.forEach(scheme => {
      const li = document.createElement("li");
      li.textContent = scheme;
      ul.appendChild(li);
    });
  } catch (error) {
    alert(error.message);
  }
});
</script>
